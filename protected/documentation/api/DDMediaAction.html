public function rcopy($src, $dst) 
{
    if (file_exists($dst)) $this->rrmdir($dst);
    if (is_dir($src)) {
        mkdir($dst);
        $files = scandir($src);
        foreach ($files as $file)
            if ($file != "." && $file != "..") $this->rcopy("$src/$file", "$dst/$file"); 
    }
